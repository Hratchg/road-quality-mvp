```typescript
// Auth API client — register, login, logout + token storage.
// Locked decisions: 04-CONTEXT.md D-04 (localStorage), D-05 (demo creds), D-06 (endpoint shapes).
// Token storage seam: getToken / clearToken are the ONLY exported surface that
// touches localStorage. setToken is private — callers MUST go through register
// or login. If we ever migrate to httpOnly cookies (post-MVP), this is the
// single file that changes.

const API_BASE = import.meta.env.VITE_API_URL || "/api";
const TOKEN_KEY = "rq_auth_token";

export interface AuthResponse {
  access_token: string;
  token_type: string;
  user_id?: number; // present on /register, absent on /login
  email?: string;   // present on /register, absent on /login
}

async function _postJson(path: string, body: unknown): Promise<Response> {
  return fetch(`${API_BASE}${path}`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body),
  });
}

export async function register(email: string, password: string): Promise<AuthResponse> {
  const res = await _postJson("/auth/register", { email, password });
  if (!res.ok) {
    const body = await res.json().catch(() => ({}));
    throw new Error(body.detail || `Register failed: ${res.status}`);
  }
  const data = (await res.json()) as AuthResponse;
  setToken(data.access_token);
  return data;
}

export async function login(email: string, password: string): Promise<AuthResponse> {
  const res = await _postJson("/auth/login", { email, password });
  if (!res.ok) {
    const body = await res.json().catch(() => ({}));
    throw new Error(body.detail || `Login failed: ${res.status}`);
  }
  const data = (await res.json()) as AuthResponse;
  setToken(data.access_token);
  return data;
}

export async function logout(): Promise<void> {
  // Server-side is a no-op for stateless JWT; we just clear localStorage.
  // Calling /auth/logout for symmetry — server returns 204, no body.
  // Swallow network errors (logout should always succeed client-side).
  try {
    await _postJson("/auth/logout", {});
  } catch {
    // intentionally ignored
  }
  clearToken();
}

export function getToken(): string | null {
  return localStorage.getItem(TOKEN_KEY);
}

function setToken(t: string): void {
  localStorage.setItem(TOKEN_KEY, t);
}

export function clearToken(): void {
  localStorage.removeItem(TOKEN_KEY);
}

export function isAuthenticated(): boolean {
  return getToken() !== null;
}
```

Critical implementation details:
1. `setToken` is a private (non-exported) function — callers MUST go through `register` or `login` to set the token. Prevents tests / ad-hoc code from injecting a fake token outside the auth flow.
2. `clearToken` IS exported (used by `api.ts`'s 401 handler). `getToken` is exported (used by `api.ts`'s header injection).
3. `isAuthenticated` is a convenience for the modal-open default state but NOT used by the network layer (which always reads `getToken` afresh on each call).
4. Error messages prefer the backend's `detail` field but fall back to a generic message if the body isn't JSON. RESEARCH §6 line 1093 verbatim.
5. `API_BASE` resolution mirrors `api.ts`'s exact pattern.

Do NOT add a `getUser()` or `/auth/me` call — RESEARCH Open Q4 says skip /auth/me for M1.

```typescript
import { getToken, clearToken } from "./api/auth";

const API_BASE = import.meta.env.VITE_API_URL || "/api";

// Thrown when a gated request hits 401, so the caller (RouteFinder) can open
// the SignInModal. Phase 4 SC #3 + 04-CONTEXT.md D-04.
export class UnauthorizedError extends Error {
  constructor() {
    super("Unauthorized");
    this.name = "UnauthorizedError";
  }
}

function authHeaders(): Record<string, string> {
  const t = getToken();
  return t ? { Authorization: `Bearer ${t}` } : {};
}

// /segments stays public per SC #3 — no auth header attached.
export async function fetchSegments(bbox: string) {
  const res = await fetch(`${API_BASE}/segments?bbox=${bbox}`);
  if (!res.ok) throw new Error(`Segments fetch failed: ${res.status}`);
  return res.json();
}

export interface RouteRequestBody {
  origin: { lat: number; lon: number };
  destination: { lat: number; lon: number };
  include_iri: boolean;
  include_potholes: boolean;
  weight_iri: number;
  weight_potholes: number;
  max_extra_minutes: number;
}

// /route is gated per SC #3 — attach auth header + handle 401.
export async function fetchRoute(body: RouteRequestBody) {
  const res = await fetch(`${API_BASE}/route`, {
    method: "POST",
    headers: { "Content-Type": "application/json", ...authHeaders() },
    body: JSON.stringify(body),
  });
  if (res.status === 401) {
    // Stale / missing / invalid token — wipe localStorage so the modal opens
    // in 'no existing session' state, not in 'we have a token but the
    // server rejected it' confusion.
    clearToken();
    throw new UnauthorizedError();
  }
  if (!res.ok) throw new Error(`Route fetch failed: ${res.status}`);
  return res.json();
}
```

Critical implementation details:
1. `fetchSegments` is UNCHANGED — no `Authorization` header, no 401 handling. SC #3 says `/segments` stays public.
2. `fetchRoute` adds `...authHeaders()` to the headers object via spread.
3. `UnauthorizedError extends Error` is a typed error class so `RouteFinder` can `if (err instanceof UnauthorizedError)` discriminate (RESEARCH §6 line 1185).
4. On 401, `clearToken()` runs BEFORE throwing.
5. The 401 check is `res.status === 401` (not `!res.ok` first then check status) so we get the typed error path before the generic-error path.

Do NOT call any function from `auth.ts` other than `getToken` and `clearToken` here.
Do NOT add a global window event — typed-error approach is simpler and just as effective for our single-modal-per-app scope.

```tsx
import { register, login } from "../api/auth";

interface SignInModalProps {
  open: boolean;
  onClose: () => void;
  onAuthSuccess: () => void;
}

type Mode = "login" | "register";

const DEMO_EMAIL = "demo@road-quality-mvp.dev"; // 04-CONTEXT.md D-05
const DEMO_PASSWORD = "demo1234";                // 04-CONTEXT.md D-05 (rotatable)

export default function SignInModal({ open, onClose, onAuthSuccess }: SignInModalProps) {
  const [mode, setMode] = useState<Mode>("login");
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  if (!open) return null;

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setLoading(true);
    setError(null);
    try {
      if (mode === "register") {
        await register(email, password);
      } else {
        await login(email, password);
      }
      onAuthSuccess();
      onClose();
    } catch (err: any) {
      setError(err.message || "Authentication failed");
    } finally {
      setLoading(false);
    }
  };

  const handleDemo = async () => {
    setEmail(DEMO_EMAIL);
    setPassword(DEMO_PASSWORD);
    setLoading(true);
    setError(null);
    try {
      await login(DEMO_EMAIL, DEMO_PASSWORD);
      onAuthSuccess();
      onClose();
    } catch (err: any) {
      setError(err.message || "Demo login failed");
    } finally {
      setLoading(false);
    }
  };

  // Backdrop click closes the modal; clicking inside the dialog does not.
  // RESEARCH §6 line 1015-1019 + PATTERNS lines 401-411 (z-[2000] above
  // MapView's z-[1000] and AddressInput's z-50).
  return (
    <div
      className="fixed inset-0 z-[2000] flex items-center justify-center bg-black/50"
      onClick={onClose}
    >
      <div
        className="bg-white rounded-lg shadow-xl p-6 w-96 max-w-full mx-4"
        onClick={(e) => e.stopPropagation()}
      >
        <h2 className="text-lg font-bold mb-4">
          {mode === "login" ? "Sign in" : "Create account"}
        </h2>

          <input
            type="email"
            placeholder="Email"
            required
            value={email}
            onChange={(e) => setEmail(e.target.value)}
            className="w-full border rounded px-3 py-2 focus:outline-none focus:ring-2 focus:ring-blue-400"
          />
          <input
            type="password"
            placeholder="Password (min 8 chars)"
            required
            minLength={8}
            value={password}
            onChange={(e) => setPassword(e.target.value)}
            className="w-full border rounded px-3 py-2 focus:outline-none focus:ring-2 focus:ring-blue-400"
          />

          <button
            type="submit"
            disabled={loading}
            className="w-full bg-blue-600 text-white rounded py-2 hover:bg-blue-700 disabled:opacity-50"
          >
            {loading ? "..." : mode === "login" ? "Sign in" : "Create account"}
          </button>
        </form>

        <button
          onClick={handleDemo}
          disabled={loading}
          className="w-full mt-3 bg-gray-100 text-gray-700 rounded py-2 hover:bg-gray-200 disabled:opacity-50"
        >
          Try as demo
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser→localStorage | The JWT lives in localStorage. JavaScript-readable storage means any XSS vulnerability gives the attacker the token. CONTEXT.md D-04 explicitly accepts this trade for MVP. |
| user→modal-form | Untrusted email + password input. Pydantic on the backend (plan 04-03) validates format; the modal's `<input type="email">` and `minLength={8}` are UX-only convenience layers. |
| frontend→backend | The Authorization header crosses this boundary. CORS is `*` for dev (Phase 5 hardens). Bearer header is sent over `fetch(...)` only to /route + /auth/* — never to /segments. |
| backdrop→panel | Backdrop click closes the modal; panel click does not. Misimplementing `stopPropagation` would let clicks inside the dialog accidentally close it. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-27 | Information Disclosure | Token leaks via XSS to `localStorage` | accept | CONTEXT.md D-04 explicitly accepts this trade. Mitigations: React's default JSX escaping (no `dangerouslySetInnerHTML` anywhere), 7-day token expiry caps blast radius, AUTH_SIGNING_KEY rotation as global revocation lever. RESEARCH Pitfall 6. |
| T-04-28 | Tampering | A malicious dependency reads `localStorage["rq_auth_token"]` | accept | Same trade as T-04-27. The `rq_` prefix on the key is a tripwire — a token under a different key suggests tampering. |
| T-04-29 | Tampering | The "Try as demo" button is hardcoded to a public credential | accept | D-05 explicitly accepts this — the demo password IS public, and the public-demo strategy depends on it. Rotation procedure: update the constant + `scripts/seed_demo_user.py` + redeploy. |
| T-04-30 | Information Disclosure | The modal's password input persists across a session via React state | mitigate | The modal unmounts when `open===false` (`if (!open) return null`), which destroys the state. Re-opening starts fresh. The password is never written to localStorage / sessionStorage. |
| T-04-31 | Information Disclosure | The "register" mode shows the previous mode's error message | mitigate | Mode toggle clears `error` (`setError(null)` on mode change). Test #4 in the human-verify checkpoint exercises this. |
