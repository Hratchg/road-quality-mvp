    /tmp/rq-venv/bin/python -c "
    import os, sys
    sys.path.insert(0, '.')
    from scripts.fetch_eval_data import _DEFAULT_LA_BBOXES
    # Adjacent expansion: union all sub-tiles into 0.01-deg parents +
    # one 0.01-deg ring of expansion around each parent zone.
    parents = {}
    for name, (mn_lon, mn_lat, mx_lon, mx_lat) in _DEFAULT_LA_BBOXES.items():
        # Strip _sw/_se/_nw/_ne suffix to get parent zone
        zone = '_'.join(name.split('_')[:-1]) if '_' in name else name
        if zone not in parents:
            parents[zone] = [mn_lon, mn_lat, mx_lon, mx_lat]
        else:
            p = parents[zone]
            p[0] = min(p[0], mn_lon); p[1] = min(p[1], mn_lat)
            p[2] = max(p[2], mx_lon); p[3] = max(p[3], mx_lat)
    # Adjacent-expansion: pad each parent by 0.005 deg on every side
    expanded = []
    for zone, (mn_lon, mn_lat, mx_lon, mx_lat) in parents.items():
        expanded.append((zone, mn_lon - 0.005, mn_lat - 0.005, mx_lon + 0.005, mx_lat + 0.005))
    print('-- D-16 re-ingest bbox-derived segment selection (training zones + adjacent expansion)')
    union_clauses = []
    for zone, mn_lon, mn_lat, mx_lon, mx_lat in expanded:
        union_clauses.append(
            f\"SELECT id FROM road_segments WHERE ST_Intersects(geom, ST_MakeEnvelope({mn_lon}, {mn_lat}, {mx_lon}, {mx_lat}, 4326))  -- {zone}\"
